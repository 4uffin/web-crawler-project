To allow the content to fit the screen automatically, do not use SizedBox(height: 600) – because this sets the height to 600 pixels on every screen, regardless of whether it’s small or large.
Instead, let the content take the available space on the screen using Expanded. Also, use SingleChildScrollView to allow scrolling when the content overflows on smaller screens.
It's should be like this:
LayoutBuilder(
builder: (context, constraints) {
return Column(
mainAxisAlignment: MainAxisAlignment.center,
children: [
Padding(
padding: const EdgeInsets.symmetric(horizontal: 30.0),
child: SizedBox(
height: 50,
child: TabBar(
controller: _tabController1,
labelColor: Colors.blue,
unselectedLabelColor: Colors.lightBlue,
indicatorColor: Colors.redAccent,
tabs: [Tab(text: "Flutter")],
),
),
),
Expanded(
child: Padding(
padding: const EdgeInsets.symmetric(horizontal: 30.0),
child: TabBarView(
controller: _tabController1,
children: [
SingleChildScrollView(
child: Column(
crossAxisAlignment: CrossAxisAlignment.start,
children: [
Padding(
padding: const EdgeInsets.symmetric(vertical: 10.0),
child: Text(
ArticlesConstants.articlesConstants2,
maxLines: 100,
style: Theme.of(context).textTheme.bodyMedium,
),
),
Text(
"Articles on FLUTTER",
maxLines: 100,
style: Theme.of(context).textTheme.bodyMedium?.copyWith(
fontWeight: FontWeight.w700,
color: Colors.redAccent,
),
),
HyperlinkText(
text: "Tic Tac Toe",
url: "",
choice: "web",
),
HyperlinkText(
text: "Tic Tac Toe: Bot vs Player - Player first move",
url: "",
choice: "web",
),
],
),
),
],
),
),
),
],
);
},
)
If you want to make the whole screen scrollable, try this code
SingleChildScrollView(
// This allows the whole screen to scroll vertically
child: Padding(
// Horizontal padding around the entire content
padding: const EdgeInsets.symmetric(horizontal: 30.0),
child: Column(
crossAxisAlignment: CrossAxisAlignment.start,
children: [
SizedBox(height: 50), // Spacing before the TabBar
// The TabBar itself
TabBar(
controller: _tabController1,
labelColor: Colors.blue,
unselectedLabelColor: Colors.lightBlue,
indicatorColor: Colors.redAccent,
tabs: [
Tab(text: "Flutter"),
],
),
SizedBox(height: 20), // Spacing between TabBar and content
// Instead of using TabBarView, manually display content based on selected tab index
Builder(
builder: (context) {
if (_tabController1.index == 0) {
// Content for the first tab
return Column(
crossAxisAlignment: CrossAxisAlignment.start,
children: [
// Top article text
Padding(
padding: const EdgeInsets.symmetric(vertical: 10.0),
child: Text(
ArticlesConstants.articlesConstants2,
maxLines: 100,
style: Theme.of(context).textTheme.bodyMedium,
),
),
// Section heading
Text(
"Articles on FLUTTER",
maxLines: 100,
style: Theme.of(context).textTheme.bodyMedium?.copyWith(
fontWeight: FontWeight.w700,
color: Colors.redAccent,
),
),
// Hyperlink 1
HyperlinkText(
text: "Tic Tac Toe",
url: "",
choice: "web",
),
// Hyperlink 2
HyperlinkText(
text: "Tic Tac Toe: Bot vs Player - Player first move",
url: "",
choice: "web",
),
// You can add more content here...
],
);
} else {
// Placeholder for other tabs (optional)
return Container();
}
},
),
SizedBox(height: 30), // Bottom spacing, in case more widgets follow
],
),
),
)
