docker-compose to run mitmproxy with web interface using docker volume
Raw
docker-compose.yml
version: '3.4'
services:
app:
image: mitmproxy/mitmproxy
ports:
- 8080:8080
- 8081:8081
volumes:
- mitmproxy:/home/mitmproxy/.mitmproxy
command: mitmweb --web-iface 0.0.0.0
volumes:
mitmproxy:
external: true
