Microsoft.PowerShell_profile.ps1
#Requires -Version 7
# Import Modules
Import-Module PSReadLine
# Import-Module oh-my-posh
oh-my-posh init pwsh --config "$env:POSH_THEMES\custom.omp.json" | Invoke-Expression
Import-Module posh-git
# Set Colors
Set-PSReadlineOption -Colors @{
"Command" = [ConsoleColor]::Green
"Parameter" = [ConsoleColor]::DarkGray
"Operator" = [ConsoleColor]::Magenta
"Variable" = [ConsoleColor]::White
"String" = [ConsoleColor]::Yellow
"Number" = [ConsoleColor]::Blue
"Type" = [ConsoleColor]::Cyan
"Comment" = [ConsoleColor]::DarkCyan
"InlinePrediction" = [ConsoleColor]::DarkGray
}
# Set Show Tooltips
Set-PSReadlineOption -ShowToolTips
# IntelliSense Prediction Source (Default: None)
Set-PSReadLineOption -PredictionSource History
# Set-PSReadLineOption -PredictionSource HistoryAndPlugin # PSReadLine ^2.2
# Set-PSReadLineOption -PredictionSource Plugin # PSReadLine ^2.2
# IntelliSense Prediction View Style (Default: List)
# Set-PSReadLineOption -PredictionViewStyle ListView # PSReadLine ^2.2
# Set-PSReadLineOption -PredictionViewStyle InlineView # PSReadLine ^2.2
# Git Status Configuration
# echo $GitPromptSettings # Show All Git Status Configurations
# $GitPromptSettings.DefaultPromptPrefix.Text = "$([char]0x2192) " # Arrow Unicode Symbol
# $GitPromptSettings.DefaultPromptSuffix.Text = "$([char]0x203A) " # Chevron Unicode Symbol
$GitPromptSettings.DefaultPromptPrefix.ForegroundColor = [ConsoleColor]::Green
$GitPromptSettings.DefaultPromptPath.ForegroundColor = [ConsoleColor]::Cyan
$GitPromptSettings.DefaultPromptSuffix.ForegroundColor = [ConsoleColor]::Cyan
$GitPromptSettings.BeforeStatus.ForegroundColor = [ConsoleColor]::Blue
$GitPromptSettings.BranchColor.ForegroundColor = [ConsoleColor]::Blue
$GitPromptSettings.AfterStatus.ForegroundColor = [ConsoleColor]::Blue
# Remove Aliases
Remove-Item alias:iex -Force
# Aliases
$devfolder = "C:\HR-DEV"
Set-Alias docker_rm_volumes RemoveAllVolumes
Set-Alias docker_rm_images RemoveAllImages
Set-Alias docker_rm_cache_builder RemoveAllBuilderCache
Set-Alias docker_rm_cache_system RemoveAllSystemCache
Set-Alias docker_rm_cache_all RemoveAllCache
Set-Alias iex Invoke-Expression
Set-Alias steamcmd "C:\Users\HerculesRw\Desktop\steamcmd\steamcmd.exe"
function RemoveAllVolumes {
param ()
$volumes = docker volume ls -q
foreach ($volumeName in $volumes) {
Write-Output "Removing volume $volumeName"
docker volume rm $volumeName
}
}
function RemoveAllImages {
param ()
$images = docker image ls -q
foreach ($imageId in $images) {
Write-Output "Removing image $imageId"
docker image rm $imageId
}
}
function RemoveAllBuilderCache {
param ()
Write-Output "Removing All Builder Cache"
docker builder prune --all
}
function RemoveAllSystemCache {
param ()
Write-Output "Removing All System Cache"
docker system prune --all
}
function RemoveAllCache {
param ()
Write-Output "Removing All Cache"
RemoveAllBuilderCache
RemoveAllSystemCache
}
