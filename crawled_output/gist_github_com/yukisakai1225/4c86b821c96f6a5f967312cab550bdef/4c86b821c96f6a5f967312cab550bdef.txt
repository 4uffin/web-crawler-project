// lib/semantics_extension.dart
import "package:flutter/material.dart";
import "package:flutter/rendering.dart";
import "package:flutter/widgets.dart";
void registerSemanticsDumpExtension() {
// Binding を初期化（WidgetBinding が無ければここで確実に作る）
final binding = WidgetsFlutterBinding.ensureInitialized();
// Semantics を有効化（release ビルドでも動くように）
SemanticsBinding.instance.ensureSemantics();
// ignore: invalid_use_of_protected_member
binding.registerServiceExtension(
name: "getSemanticsInfo",
callback: (parameters) async {
// 現在のFlutterバージョンでは、SemanticsOwnerからrootを取得
final root = WidgetsBinding
.instance.pipelineOwner.semanticsOwner?.rootSemanticsNode;
if (root == null) {
return {"result": []};
}
final list = <Map<String, dynamic>>[];
void collect(SemanticsNode node) {
final d = node.getSemanticsData();
// 意味のあるノードだけ取り出す
final hasLabel =
d.label.isNotEmpty || d.value.isNotEmpty || d.hint.isNotEmpty;
final hasAction = d.actions != 0;
if (hasLabel || hasAction) {
final globalRect = MatrixUtils.transformRect(
node.transform ?? Matrix4.identity(),
node.rect,
);
var ancestorTopLeft = Offset.zero;
var parent = node.parent;
final ancestorTopLefts = <Offset>[];
while (parent != null) {
final parentRectTopLeft = MatrixUtils.transformPoint(
parent.transform ?? Matrix4.identity(),
parent.rect.topLeft,
);
ancestorTopLeft = ancestorTopLeft + parentRectTopLeft;
ancestorTopLefts.add(ancestorTopLeft);
parent = parent.parent;
}
list.add({
"label": d.label,
"value": d.value,
"hint": d.hint,
"actions": _actionsToStrings(d.actions),
"rect": {
"left": globalRect.left + ancestorTopLeft.dx,
"top": globalRect.top + ancestorTopLeft.dy,
"width": globalRect.width,
"height": globalRect.height,
},
});
}
// 現在のFlutterバージョンでは、getChildrenメソッドを使用
node.visitChildren((child) {
collect(child);
return true; // 続けて子ノードを訪問
});
}
collect(root);
return {"result": list};
},
);
}
/// SemanticsAction → 可読文字列
List<String> _actionsToStrings(int actions) {
const mapping = <SemanticsAction, String>{
SemanticsAction.tap: "tap",
SemanticsAction.longPress: "longPress",
SemanticsAction.scrollLeft: "scrollLeft",
SemanticsAction.scrollRight: "scrollRight",
SemanticsAction.scrollUp: "scrollUp",
SemanticsAction.scrollDown: "scrollDown",
SemanticsAction.increase: "increase",
SemanticsAction.decrease: "decrease",
// 必要に応じて追加
};
return mapping.keys
.where((a) => actions & a.index != 0)
.map((a) => mapping[a]!)
.toList(growable: false);
}
