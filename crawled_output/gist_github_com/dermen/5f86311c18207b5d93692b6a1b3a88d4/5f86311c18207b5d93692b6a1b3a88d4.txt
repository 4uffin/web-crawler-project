simulate two orientations of the same crystal with nanoBragg+simtbx
Raw
quick_nanobragg.py
import numpy as np
from scipy.spatial.transform import Rotation
import pylab as plt
from dxtbx.model import Crystal
from dxtbx.model.beam import BeamFactory
from scitbx.matrix import sqr
from simtbx import nanoBragg
from simtbx.nanoBragg.sim_data import Amatrix_dials2nanoBragg, SimData
from simtbx.nanoBragg.nanoBragg_beam import NBbeam
from simtbx.nanoBragg.nanoBragg_crystal import NBcrystal
from simtbx.diffBragg import utils
cuda=False
# make structure factors
F = utils.get_complex_fcalc_from_pdb("MYO-spars_refine_046_nowater.pdb").as_amplitude_array()
# make dxtbx crystal and beam
dxtbx_det = SimData.simple_detector(detector_distance_mm=200, pixelsize_mm=0.1, image_shape=(2500,2500))
dxtbx_beam = BeamFactory.simple(wavelength=1)
# make the dxtbx crystal from the pdb file:
uc = F.crystal_symmetry().unit_cell()
B = np.reshape(uc.orthogonalization_matrix(), (3,3))
# note, a,b,c are real space unit cell vectors, I never know the convention, its either B.T or B. for
# some space group, its irrelevant:
a,b,c = B.T
sym = F.crystal_symmetry().space_group().type().lookup_symbol()
dxtbx_cryst = Crystal(a,b,c,sym)
# make high level nanoBragg crystal
cryst = NBcrystal()
cryst.dxtbx_crystal = dxtbx_cryst
cryst.miller_array = F
# make high level nanoBragg beam
beam = NBbeam()
beam.unit_s0 = dxtbx_beam.get_unit_s0()
beam.spectrum = [(dxtbx_beam.get_wavelength(), 1e12)]
# make simulator:
S = SimData()
S.detector = dxtbx_det
S.crystal = cryst
S.beam = beam
S.instantiate_nanoBragg()
# Now, the object S.D is a simulator (an instance of nanoBragg)
# simulate orientation
if cuda:
S.D.add_nanoBragg_spots_cuda()
else:
S.D.add_nanoBragg_spots()
img = S.D.raw_pixels.as_numpy_array()
# reset pixel values , rotate crystal, and re-simulate:
S.D.raw_pixels*=0
new_U = tuple(Rotation.random(1).as_matrix().ravel())
dxtbx_cryst.set_U(new_U)
S.D.Amatrix = sqr(dxtbx_cryst.get_A()).transpose()
if cuda:
S.D.add_nanoBragg_spots_cuda()
else:
S.D.add_nanoBragg_spots()
img2 = S.D.raw_pixels.as_numpy_array()
# display
m = img.mean()
s = img.std()
fig, (ax,ax2) = plt.subplots(1,2)
ax.imshow(img,vmin=0, vmax=m+s)
ax.set_title("Umat is identity")
ax2.imshow(img2,vmin=0, vmax=m+s)
ax2.set_title("Umat is randomized")
plt.show()
