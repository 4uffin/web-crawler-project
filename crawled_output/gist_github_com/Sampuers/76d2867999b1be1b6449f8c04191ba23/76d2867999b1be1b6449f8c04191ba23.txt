import pandas as pd
import matplotlib.pyplot as plt
print(" An√°lisis de Gastos ")
print("=" * 40)
# Leer los datos
df = pd.read_csv("gastos.csv", parse_dates=["fecha"])
# Crear columna de mes
df["mes"] = df["fecha"].dt.to_period("M")
# Resumen mensual por categor√≠a
resumen = df.groupby(["mes", "categoria"])["importe"].sum().unstack().fillna(0)
print("Resumen mensual por categor√≠a:")
print(resumen)
# Crear gr√°fico
resumen.plot(kind="bar", stacked=True)
plt.title("Gastos mensuales por categor√≠a")
plt.ylabel("‚Ç¨")
plt.tight_layout()
# Guardar gr√°fico
plt.savefig("resumen_gastos.png")
print("Gr√°fico guardado como 'resumen_gastos.png'")
# Mostrar gr√°fico
plt.show()
print("An√°lisis completado!")
